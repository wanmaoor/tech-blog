class EventBus {
  constructor() {
    this.events = {};
  }

  on(msg, fn) {
    if (!this.events[msg]) {
      this.events[msg] = [];
    }
    this.events[msg].push(fn);
  }

  trigger(msg, params) {
    this.events[msg].forEach(item => {
      item.apply(this, params);
    });
  }

  off(msg) {
    if (!!this.events[msg]) {
      delete this.events[msg];
      console.log(`DELETE ${msg} COMPLETED!`);
    } else {
      console.error("CAN'T DELETE PROPERTY OF UNDEFINED");
    }
  }
}

export default EventBus;
